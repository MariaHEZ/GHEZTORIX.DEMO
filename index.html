let usuarios = [ { id: 1, nombre: "Admin", correo: "admin@gheztorix.com", password: "12345", rol: "admin" } ]; let clientes = [ { id: 1, nombre: "Cliente Demo", rfc: "ABC123", password: "123", documentos: [], comentarios: [] } ]; function login(){ let user = document.getElementById("usuario").value; let pass = document.getElementById("password").value; // ADMIN let admin = usuarios.find(u => u.correo === user && u.password === pass); if(admin){ localStorage.setItem("sesion", JSON.stringify(admin)); location.href="panel.html"; return; } // CLIENTE let cliente = clientes.find(c => c.rfc === user && c.password === pass); if(cliente){ localStorage.setItem("sesion", JSON.stringify({ rol:"cliente", data:cliente })); location.href="cliente.html"; return; } alert("Credenciales incorrectas"); } let sesion = JSON.parse(localStorage.getItem("sesion")); if(!sesion || sesion.rol !== "admin"){ location.href="login.html"; } function renderClientes(){ let html = ""; clientes.forEach((c,i)=>{ html += `
${c.nombre}
`; }); document.getElementById("panel").innerHTML = html; } function eliminar(i){ clientes.splice(i,1); guardar(); renderClientes(); } function editar(i){ let nuevo = prompt("Nuevo nombre:", clientes[i].nombre); if(nuevo){ clientes[i].nombre = nuevo; guardar(); renderClientes(); } } function guardar(){ localStorage.setItem("clientes", JSON.stringify(clientes)); } function subirDocumento(i){ let doc = prompt("Nombre del documento"); clientes[i].documentos.push({ nombre: doc, fecha: new Date().toLocaleString() }); guardar(); } function verDocs(i){ let html = "
