const USER = 'admin@gheztorix.com'; const PASS = '12345'; // ================= BASE DE DATOS ================= let clientes = JSON.parse(localStorage.getItem('clientes_gz')) || []; let clienteActivo = null; let moduloActivo = null; // ================= LOGIN ================= function login(){ // leer inputs correctamente (IMPORTANTE) const correo = document.getElementById('usuario').value.trim(); const password = document.getElementById('password').value.trim(); if(correo === USER && password === PASS){ document.getElementById('loginScreen').classList.add('hidden'); document.getElementById('dashboard').classList.remove('hidden'); document.getElementById('error').style.display = 'none'; renderStats(); } else { document.getElementById('error').style.display = 'block'; } } // ================= LOGOUT ================= function logout(){ document.getElementById('dashboard').classList.add('hidden'); document.getElementById('loginScreen').classList.remove('hidden'); } // ================= GUARDAR CLIENTE ================= function guardarCliente(){ const c = { id: Date.now(), nombre: document.getElementById('cliente').value.trim(), rfc: document.getElementById('rfc').value.trim(), pass: document.getElementById('passCliente').value.trim(), datos:{ documentos:[], facturas:[], declaraciones:[], impuestos:[], cuentas:[], reportes:[] } }; if(!c.nombre || !c.rfc || !c.pass){ alert("Completa todos los campos"); return; } clientes.push(c); persist(); alert("Cliente registrado"); limpiarCliente(); renderStats(); } // ================= LIMPIAR ================= function limpiarCliente(){ document.getElementById('cliente').value = ''; document.getElementById('rfc').value = ''; document.getElementById('passCliente').value = ''; } // ================= BUSCAR CLIENTE ================= function buscarCliente(){ const rfc = document.getElementById('buscarRFC').value.trim(); const pass = document.getElementById('buscarPASS').value.trim(); const r = clientes.find(c => c.rfc === rfc && c.pass === pass); if(!r){ alert("Cliente no encontrado"); return; } clienteActivo = r; alert("Accediendo a: " + clienteActivo.nombre); } // ================= ABRIR MÓDULO ================= function abrirModulo(m){ moduloActivo = m; const lista = clienteActivo.datos[m] || []; const tabla = document.getElementById('tabla'); tabla.innerHTML = ''; lista.forEach((x,i)=>{ tabla.innerHTML += ` ${x} `; }); } // ================= AGREGAR DATO ================= function guardarModulo(){ const input = document.getElementById('registro'); if(!clienteActivo.datos[moduloActivo]){ clienteActivo.datos[moduloActivo] = []; } clienteActivo.datos[moduloActivo].push(input.value); persist(); abrirModulo(moduloActivo); input.value = ''; } // ================= EDITAR ================= function editar(i){ let nuevo = prompt("Editar registro:"); if(nuevo){ clienteActivo.datos[moduloActivo][i] = nuevo; persist(); abrirModulo(moduloActivo); } } // ================= ELIMINAR ================= function eliminar(i){ clienteActivo.datos[moduloActivo].splice(i,1); persist(); abrirModulo(moduloActivo); } // ================= GUARDAR EN LOCALSTORAGE ================= function persist(){ localStorage.setItem('clientes_gz', JSON.stringify(clientes)); } // ================= ESTADÍSTICAS ================= function renderStats(){ document.getElementById('sTotal').textContent = clientes.length; }
