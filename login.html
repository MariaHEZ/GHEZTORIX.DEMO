const usuarios = [ {usuario:"admin", password:"123456", rol:"admin"}, {usuario:"cliente1", password:"123456", rol:"cliente"} ]; function login(){ let usuario = document.getElementById("usuario").value; let password = document.getElementById("password").value; let acceso = usuarios.find(u => u.usuario === usuario && u.password === password ); if(acceso){ // guardar sesión completa localStorage.setItem("sesion", JSON.stringify(acceso)); // SIEMPRE entra al mismo sistema window.location.href = "dashboard.html"; }else{ document.getElementById("error").innerText = "Usuario o contraseña incorrectos"; } }
